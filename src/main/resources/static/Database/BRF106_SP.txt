------------------BRF106_DETAIL_PROCEDURE-------

create or replace PROCEDURE BRF106_DETAIL_PROCEDURE (REPORT_DATE DATE)
IS
  V_REPORT_DATE DATE := To_date (REPORT_DATE, 'DD-MM-YYYY'); 
   V_FROM_DATE dATE :=  ADD_MONTHS (LAST_DAY (REPORT_DATE), -1) + 1;
BEGIN


delete from BRF106_DETAILTABLE where report_date = V_REPORT_DATE;
Commit;


insert into  brf106_mapping_table
( foracid, cust_id, ACCT_NAME,GL_SUB_HEAD_CODE, SCHM_CODE,NRE_STATUS,ACCT_CRNCY_CODE,REPORT_DATE,
ACT_BALANCE_AMT_LC,SOL_ID,CUST_TYPE,CONSTITUTION_CODE,ACID)
SELECT  ACCT_NUMBER, cust_id, ACCT_NAME,GL_SUB_HEAD_CODE, SCHM_CODE,NRE_FLG,ACCT_CRNCY_CODE,REPORT_DATE,ACT_BALANCE_AMT_LC,
SOL_ID,CUST_TYPE_CODE_2,CONSTITUTION_CODE,ACID
FROM GENERAL_MASTER_TABLE  where report_date = '30-JUN-2023'AND Constitution_Code IN ('11501', '11502', '11503', '11504', '11505', '11506', 
'11509', '11599', '11601', '11602', '11603', '11604', '11605', '11606', '11607', '11608', '11699', '22101', '22201', '22202', '22301', '22302', '22303', '22304', '22305', '22401', 
'22402', '22403', '22404', '33101', '33102', '33103', '33104', '33199', '33201', '33202', '33203', '33204', '33299', '45507', '56500', 
'56600', '56601', '56700', '61001', '61002', '61003', '61004', '61006', '61007', '61008', '61010', '61011', '61012', '61013', '71001', 
'71002', '71004', '71008', '71009', '71010', '71011');
COMMIT;

MERGE INTO BRF106_DETAILTABLE M
USING (
    SELECT   b.REPORT_LABEL_1,b.cust_id,b.foracid
    FROM brf106_mapping_table b
    JOIN BRF106_DETAILTABLE a ON a.foracid = b.foracid
    WHERE a.report_date = V_REPORT_DATE
) S
ON (M.foracid = S.foracid)
WHEN MATCHED THEN UPDATE
SET M.REPORT_LABEL_1 = S.REPORT_LABEL_1
WHERE M.report_date = V_REPORT_DATE;
COMMIT;

MERGE INTO BRF106_DETAILTABLE M
USING (
    SELECT   b.nre_status,b.cust_id,b.foracid
    FROM brf106_mapping_table b
    JOIN BRF106_DETAILTABLE a ON a.foracid = b.foracid
    WHERE a.report_date = V_REPORT_DATE
) S
ON (M.foracid = S.foracid)
WHEN MATCHED THEN UPDATE
SET M.NRE_FLG = S.nre_status
WHERE M.report_date = V_REPORT_DATE;
COMMIT;


END;
