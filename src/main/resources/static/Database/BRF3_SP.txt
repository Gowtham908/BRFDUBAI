--------BRF3_DETAIL_PROCEDURE-----------------
create or replace PROCEDURE BRF3_DETAIL_PROCEDURE (REPORT_DATE DATE)
IS
   v_report_date   date := to_date (report_date, 'DD-MM-YYYY');
BEGIN


delete from BRF3_DETAILTABLE where report_date = V_REPORT_DATE;
Commit;


insert into BRF3_DETAILTABLE
( foracid, cust_id, ACCT_NAME,REPORT_DATE,NRE_FLG, REPORT_LABEL_1,REPORT_NAME_1,ACCT_CRNCY_CODE,REPORT_ADDL_CRITERIA_1)
SELECT  foracid, CUST_ID, ACCT_NAME,REPORT_DATE,NRE_FLG,REPORT_LABEL_1,REPORT_NAME_1,ACCT_CRNCY_CODE, REPORT_ADDL_CRITERIA_1 FROM 
brf3_mapping_table where report_date = '30-jun-2023' AND REPORT_LABEL_1 IS NOT NULL ;


MERGE INTO BRF3_DETAILTABLE A
USING ( SELECT ACCT_NUMBER, ACT_BALANCE_AMT_LC FROM GENERAL_MASTER_TABLE) B
ON (A.foracid = B.ACCT_NUMBER)
WHEN MATCHED THEN
  UPDATE SET A.ACT_BALANCE_AMT_LC = B.ACT_BALANCE_AMT_LC;
  COMMIT;

COMMIT;


END;
