create or replace PROCEDURE         "BRF46_DETAIL_PROCEDURE" (REPORT_DATE DATE)
IS
  V_REPORT_DATE DATE := To_date (REPORT_DATE, 'DD-MM-YYYY'); 

BEGIN


delete from BRF46_DETAILTABLE where REPORT_DATE = V_REPORT_DATE;
Commit;


insert into BRF46_DETAILTABLE
( foracid, cust_id, ACCT_NAME,REPORT_DATE,REPORT_LABEL_1,REPORT_NAME_1,REPORT_ADDL_CRITERIA_1)
SELECT  foracid, cust_id, ACCT_NAME,REPORT_DATE,REPORT_LABEL_1,REPORT_NAME_1,REPORT_ADDL_CRITERIA_1 FROM 
brf46_mapping_table where REPORT_DATE = '30-JUN-2023' AND REPORT_LABEL_1 IS NOT NULL ;

MERGE INTO BRF46_DETAILTABLE A
USING ( SELECT ACCT_NO, AMT_LC FROM BRF_TREASURY_MASTER_TB) B
ON (A.foracid = B.ACCT_NO)
WHEN MATCHED THEN
  UPDATE SET A.ACT_BALANCE_AMT_LC = B.AMT_LC;
  COMMIT;

delete from BRF46_DETAILTABLE where REPORT_DATE = V_REPORT_DATE AND report_label_1 is null;
COMMIT;

COMMIT;
END;
